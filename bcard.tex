\documentclass[10pt,oneside,final]{article}

\def\yesbleed{}
\def\yesbleedguide{}

\usepackage{calc} % needed for math
\newlength{\width}\setlength{\width}{3.5in}
\newlength{\height}\setlength{\height}{2in}
\ifdefined\yesbleed
    \newlength{\bleed}\setlength{\bleed}{0.125in}
\else
    \newlength{\bleed}\setlength{\bleed}{0in}
\fi

% set all margins to 0 and set business card size
\usepackage[
    paperwidth=\width+2\bleed,
    paperheight=\height+2\bleed,
    margin=0cm,
    noheadfoot,
]{geometry}
\setlength{\baselineskip}{0cm}
\setlength{\topskip}{0pt}

% times roman, including math (where possible)
\usepackage{mathptmx}
% hopefully embeds all fonts in pd
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{pslatex}
% more compact and arguably nicer
\usepackage[kerning,spacing]{microtype}

\usepackage{parskip}        % remove paragraph indents
\usepackage{tikz}           % drawing
\usepackage{xcolor}         % more colour options

% qr code things
\usepackage{graphics}
\usepackage[nolinks]{qrcode}


% load and configure tikz libraries
\usetikzlibrary{
    matrix,
    calc,
    positioning,
}

% define some lengths for internal spacing
\newlength{\icoscale}\setlength{\icoscale}{3pt}
\newlength{\icopad}\setlength{\icopad}{4.5\icoscale}
\newlength{\descpad}\setlength{\descpad}{4.5\icoscale}
\newlength{\descsep}\setlength{\descsep}{2pt}

% colour options
\definecolor{bgi}{HTML}{e4bc7f}
\definecolor{bgii}{HTML}{da8447}

% change global colour
%\makeatletter
%\newcommand{\globalcolor}[1]{%
%  \color{#1}\global\let\default@color\current@color
%}
%\makeatother
%\AtBeginDocument{\globalcolor{textcolour}}

\begin{document}

    \thispagestyle{empty}
    % uncomment to center vertically
    %\vspace*{\fill}
    \begin{center}
    \begin{tikzpicture}
        % hacky, but this will cover the whole page
        \node(bg) [
            %fill=blue,
            minimum width=\width+2*\bleed,
            minimum height=\height+2*\bleed,
        ] {};

        % add a bit of color
        \node(bgtop) [
            fill=bgi,
            minimum width=\width+2*\bleed,
            %minimum height=0.382*\height,
            minimum height=0.333*\height+\bleed+\icoscale,
            anchor=north west,
        ] at (bg.north west) {};

        % this defines the card geometry, ignoring bleeds
        \node(card) [
            minimum width=\width,
            minimum height=\height,
            anchor=north west,
        ] at ($(bg.north west)+(\bleed,-\bleed)$) {};

        % some bleed guides to help editing
        \ifdefined\yesbleedguide
            \node [
                draw,
                minimum width=\width,
                minimum height=\height,
                anchor=north west,
            ] at (card.north west) {};
            \node [
                draw, dashed,
                minimum width=\width-2*\bleed,
                minimum height=\height-2*\bleed,
                anchor=north west,
            ] at ($(card.north west)+(\bleed,-\bleed)$) {};
        \fi

        % main ico
        \node(ico) [
            anchor=west,
            %draw,
        ] at ($(card.west)+(\icopad-\icoscale,0)$) {%
            \def\littlefsicoscale{\icoscale}%
            \input{littlefs-ico}%
        };

        % name and stuff
        \node(name) [
            anchor=south east,
            align=left,
            %draw,
            inner sep=0,
            inner ysep=\descsep,
            minimum width=0,
            minimum height=0,
        ] at ({$(card.east)-(\descpad,0)$} |- bgtop.south) {
            {\Large \bf Christopher Haster}%
        };
        \node(desc) [
            anchor=north west,
            align=left,
            %draw,
            inner xsep=0,
            inner ysep=\descsep,
            minimum width=0,
            minimum height=0,
        ] at (name.south west) {
            Software Research Engineer
        };
        \node(stuff) [
            anchor=north east,
            align=right,
            %draw,
            inner xsep=0,
            inner ysep=\descsep,
            minimum width=0,
            minimum height=0,
        ] at (desc.south -| name.east) {
            %${\leftarrow}$ Made that thing\\
            geky@geky.net \\
            github.com/geky \\
            linkedin.com/in/geky
        };
    \end{tikzpicture}
    \end{center}
    % uncomment to center vertically
    %\vspace*{\fill}
